(function(){
  "use strict";

  var API = "", token = "", ME = null, _es = null, _alertsTimer = null, lastQR = null;

  // ---------- tiny helpers ----------
  function $(id){ return document.getElementById(id); }
  function auth(){ return token ? {"Authorization":"Bearer "+token} : {}; }
  function setTxt(id,txt){ var e=$(id); if(e) e.textContent = txt; }
  function showMsg(id,cls,txt){ var e=$(id); if(e){ e.className = cls || ""; e.textContent = txt; } }
  function handle(r){ return r.json().then(function(j){ if(!r.ok) throw new Error((j && j.error) || ("HTTP " + r.status)); return j; }); }
  function mustJson(r){ var ct=(r.headers.get("content-type")||"").toLowerCase(); if(ct.indexOf("application/json")===-1){ throw new Error("Wrong API ("+r.url+"): not JSON"); } return r; }
  function esc(s){ return String(s||"").replace(/[&<>"']/g, m=>({"&":"&amp;","<":"&lt;",">":"&gt;","\"":"&quot;","'":"&#39;"}[m])); }
  function computeApi(){ return (localStorage.getItem("ernosApi") || location.origin).replace(/\/+$/,''); }
function showLogin(){
  $("viewLogin").style.display = "";
  $("viewApp").style.display = "none";
}
function showApp(){
  $("viewLogin").style.display = "none";
  $("viewApp").style.display = "";
}
  async function apiGet(path){
    const api = computeApi();
    const jwt = localStorage.getItem("ernosToken") || "";
    const r = await fetch(api + path, { headers: jwt ? {"Authorization":"Bearer "+jwt} : {} });
    const j = await r.json().catch(()=> ({}));
    if(!r.ok) throw new Error(j.error || ("HTTP " + r.status));
    return j;
  }

  async function updateMeBadge(){
    try{
      const me = await apiGet("/me");
      const el = $("meBadge");
      if (el) el.textContent = (me.name || me.email) + " · " + (me.role || "") + " · " + (me.tenant_name || "No Site");
      return me;
    }catch(_){
      const el = $("meBadge"); if (el) el.textContent = "Not signed in";
      return null;
    }
  }

  // ---------- date/time helpers ----------
  function parseDateISO(s){
    if(!s) return null;
    let t = String(s).trim();
    if (/[zZ]$/.test(t) || /[+\-]\d{2}:?\d{2}$/.test(t)) {
      // already tz'd
    } else if (t.includes("T")) {
      t += "Z";
    } else {
      t = t.replace(" ","T") + "Z";
    }
    const d = new Date(t);
    return isNaN(d) ? null : d;
  }
  function toLocalDate(s){
    const d = parseDateISO(s); if(!d) return "";
    const dd = String(d.getDate()).padStart(2,"0");
    const mm = String(d.getMonth()+1).padStart(2,"0");
    const yy = String(d.getFullYear()).slice(-2);
    return dd + "/" + mm + "/" + yy;
  }
  function toLocalTime(s){
    const d = parseDateISO(s); if(!d) return "";
    return d.toLocaleTimeString([], {hour:"2-digit", minute:"2-digit", hour12:false});
  }
  function durationMin(a,b){
    const sd = parseDateISO(a), ed = parseDateISO(b);
    if(!sd || !ed) return "";
    return String(Math.max(0, Math.round((ed - sd) / 60000)));
  }

  function normalizeToken(raw){
    var t = String(raw||"").trim();
    try{ t = decodeURIComponent(t); }catch(_){}
    if (t.startsWith("#")) t = t.slice(1);
    if (/^https?:\/\//i.test(t)) { var parts=t.split("#"); t = parts[1] || t.split("/").pop(); }
    if (/^ci\//i.test(t)) t = t.slice(3);
    if (/^reception\//i.test(t)) t = t.slice(10);
    if (/^env\//i.test(t)) t = t.slice(4);
    if (t.indexOf("/") >= 0) t = t.split("/").pop();
    return t;
  }

function loadStaff(){
  return fetch(API + "/users", { headers: auth() })
    .then(mustJson).then(handle)
    .then(function(j){
      const tb = $("staffBody"); if (!tb) return;
      tb.innerHTML = "";
      (j.items || []).forEach(function(u){
        const tr = document.createElement("tr");
        tr.innerHTML =
          "<td>"+esc(u.name||"")+"</td>"+
          "<td>"+esc(u.email||"")+"</td>"+
          "<td>"+esc(u.role||"")+"</td>"+
          "<td>"+esc(u.category||"")+"</td>"+
          "<td>"+esc(u.title||"")+"</td>"+
          "<td><button class='small secondary e'>Edit</button> <button class='small d'>Delete</button></td>";
        tb.appendChild(tr);

        // ---- Edit ----
        const btnE = tr.querySelector("button.e");
        if (btnE) btnE.onclick = function(){
          var name = prompt("Name", u.name||""); if (name==null) return;
          var role = prompt("Role (USER/ADMIN/ADMIN_GLOBAL)", u.role||""); if (role==null) return;
          var category = prompt("Category (NURSING/HOUSEKEEPING/RECEPTION/AUDITOR/MAINTENANCE)", u.category||""); if (category==null) return;
          var title = prompt("Title", u.title||""); if (title==null) return;

          fetch(API + "/users/" + u.id, {
            method: "PATCH",
            headers: Object.assign({"Content-Type":"application/json"}, auth()),
            body: JSON.stringify({ name: name, role: role, category: category, title: title })
          })
          .then(mustJson).then(handle)
          .then(function(){ showMsg("staffMsg","ok","Updated."); return loadStaff(); })
          .catch(function(e){ showMsg("staffMsg","error", e.message); });
        };

        // ---- Delete ----
        const btnD = tr.querySelector("button.d");
        if (btnD) btnD.onclick = function(){
          if (!confirm('Delete "' + (u.name || u.email || "user") + '"?')) return;

          // Primary: modern DELETE
          fetch(API + "/users/" + u.id, { method: "DELETE", headers: auth() })
            .then(mustJson).then(handle)
            .then(function(){ showMsg("staffMsg","ok","Deleted."); return loadStaff(); })
            .catch(function(e1){
              // Fallback: legacy POST alias
              fetch(API + "/users/delete", {
                method: "POST",
                headers: Object.assign({"Content-Type":"application/json"}, auth()),
                body: JSON.stringify({ id: u.id })
              })
              .then(mustJson).then(handle)
              .then(function(){ showMsg("staffMsg","ok","Deleted."); return loadStaff(); })
              .catch(function(e2){ showMsg("staffMsg","error", e2.message || e1.message); });
            });
        };
      });
    })
    .catch(function(e){ showMsg("staffMsg","error", e.message); });
}
